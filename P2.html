def dataframe_fillvalues(df,gender):
    index=['<=10','11-20','21-30','31-40','41-50','51-60','61-70','>70']
    columns=['Not Survived Count','Survived Count']
    df_to_fill=pd.DataFrame(index=index,columns=columns)
    pd.to_numeric(df_to_fill['Survived Count'],errors='coerce')
    pd.to_numeric(df_to_fill['Not Survived Count'], errors='coerce')
    df_to_fill['Survived Count']=0
    df_to_fill['Not Survived Count']=0
    
    for i,row in df.iterrows():
        if (row['Age']<=10 and row['Sex']==gender):
            if row['Survived']==1:
                df_to_fill.loc['<=10','Survived Count'] = df_to_fill.loc['<=10','Survived Count'] + 1
            else:
                df_to_fill.loc['<=10','Not Survived Count'] = df_to_fill.loc['<=10','Not Survived Count'] + 1
        elif (row['Age']>10 and row['Age']<=20 and row['Sex']==gender):
            if row['Survived']==1:
                df_to_fill.loc['11-20','Survived Count'] = df_to_fill.loc['11-20','Survived Count'] + 1
            else:
                df_to_fill.loc['11-20','Not Survived Count'] = df_to_fill.loc['11-20','Not Survived Count'] + 1
        elif (row['Age']>20 and row['Age']<=30 and row['Sex']==gender):
            if row['Survived']==1:
                 df_to_fill.loc['21-30','Survived Count'] =  df_to_fill.loc['21-30','Survived Count'] + 1
            else:
                 df_to_fill.loc['21-30','Not Survived Count'] =  df_to_fill.loc['21-30','Not Survived Count'] + 1
        elif (row['Age']>30 and row['Age']<=40 and row['Sex']==gender):
            if row['Survived']==1:
                df_to_fill.loc['31-40','Survived Count'] = df_to_fill.loc['31-40','Survived Count'] + 1
            else:
                df_to_fill.loc['31-40','Not Survived Count'] = df_to_fill.loc['31-40','Not Survived Count'] + 1
        elif (row['Age']>40 and row['Age']<=50 and row['Sex']==gender):
            if row['Survived']==1:
                df_to_fill.loc['41-50','Survived Count'] = df_to_fill.loc['41-50','Survived Count'] + 1
            else:
                df_to_fill.loc['41-50','Not Survived Count'] = df_to_fill.loc['41-50','Not Survived Count'] + 1
        elif (row['Age']>50 and row['Age']<=60 and row['Sex']==gender):
            if row['Survived']==1:
                df_to_fill.loc['51-60','Survived Count'] = df_to_fill.loc['51-60','Survived Count'] + 1
            else:
                df_to_fill.loc['51-60','Not Survived Count'] = df_to_fill.loc['51-60','Not Survived Count'] + 1
        elif (row['Age']>60 and row['Age']<=70 and row['Sex']==gender):
            if row['Survived']==1:
                df_to_fill.loc['61-70','Survived Count'] = df_to_fill.loc['61-70','Survived Count'] + 1        
            else:
                df_to_fill.loc['61-70','Not Survived Count'] = df_to_fill.loc['61-70','Not Survived Count'] + 1
        elif (row['Age']>70 and row['Sex']==gender):
            if row['Survived']==1:
                df_to_fill.loc['>70','Survived Count'] = df_to_fill.loc['>70','Survived Count'] + 1
            else:
                df_to_fill.loc['>70','Not Survived Count'] = df_to_fill.loc['>70','Not Survived Count'] + 1
    return df_to_fill

def sum_count(column_to_look, value_to_look, df):
    sum=0.0
    for i,row in df.iterrows():
        if (row[column_to_look]==value_to_look):
            sum = sum + row['Count']
    return sum

def add_percentage(df,column_to_look):
    for i,row in df.iterrows():
        sumcount=sum_count(column_to_look, row[column_to_look],df)
        df.loc[i,'Percent Survived']=(row['Count']/sumcount) * 100
    return df

def z_test_two_proportions(p1,p2,n1,n2):
    p = (p1 * n1 + p2 * n2) / (n1 + n2)
    SE = np.sqrt( p * ( 1 - p ) * ( (1/n1) + (1/n2) ) )
    z = (p1 - p2) / SE
    print z
    z_critical=1.65
    if z > z_critical:
        print 'Test significant at p < 0.05'
    else:
        print 'Not Significant'
